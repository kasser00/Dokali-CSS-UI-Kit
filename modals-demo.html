Basic Transitions

Slide Animations

HTML
<!-- Slide Up -->
<div class="modal-overlay modal-slide-up" id="modal-slide-up">
  <div class="modal">
    <div class="modal-header">
      <h3 class="modal-title">Slide Up Modal</h3>
      <button class="modal-close" onclick="closeModal(this)">×</button>
    </div>
    <div class="modal-body">
      <p>Modal content here.</p>
    </div>
    <div class="modal-footer">
      <button class="btn btn-secondary" onclick="closeModal(this)">Cancel</button>
      <button class="btn btn-primary" onclick="closeModal(this)">Confirm</button>
    </div>
  </div>
</div>
